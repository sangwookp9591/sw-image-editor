export function useBgRemoval(fabricRef: RefObject<FabricCanvas | null>) {
  const isProcessing = useEditorStore((s) => s.isProcessing);
  const setIsProcessing = useEditorStore((s) => s.setIsProcessing);
  const setCanvasJson = useEditorStore((s) => s.setCanvasJson);

  const handleRemoveBg = useCallback(async () => {
    const canvas = fabricRef.current;
    if (!canvas || isProcessing) return;
    setIsProcessing(true);
    try {
      const savedVpt = canvas.viewportTransform;
      canvas.viewportTransform = [1, 0, 0, 1, 0, 0];
      const base64 = canvas.toDataURL({ format: "png", multiplier: 1 });
      canvas.viewportTransform = savedVpt;
      canvas.requestRenderAll();
      // ... call server action, load result, replace canvas, setCanvasJson, toast
    } catch (err) {
      toast.error(err instanceof Error ? err.message : "Failed");
    } finally {
      setIsProcessing(false);
    }
  }, [fabricRef, isProcessing, setIsProcessing, setCanvasJson]);
  return { handleRemoveBg, /* ... */ };
}

Export `function useUpscale(fabricRef: RefObject<FabricCanvas | null>)`:
- Read `isProcessing`, `setIsProcessing`, `setCanvasJson` from useEditorStore
- Define `handleUpscale = useCallback(async (scale: 2 | 4) => { ... })`:
  - Guard: `if (!canvas || isProcessing) return;`
  - `setIsProcessing(true)`
  - Save viewport: `const savedVpt = canvas.viewportTransform;`
  - Reset viewport: `canvas.viewportTransform = [1, 0, 0, 1, 0, 0];`
  - Export: `const base64 = canvas.toDataURL({ format: "png", multiplier: 1 });`
  - Restore viewport: `canvas.viewportTransform = savedVpt; canvas.requestRenderAll();`
  - Call: `const { cdnUrl } = await upscaleImage(base64, scale);`
  - Load result: `const fabric = await import("fabric"); const img = await fabric.FabricImage.fromURL(cdnUrl, { crossOrigin: "anonymous" });`
  - Scale to canvas: use `canvas.getWidth()` / `canvas.getHeight()` with aspect ratio comparison (same as use-bg-removal.ts lines 73-81)
  - Replace canvas: `canvas.clear(); canvas.add(img); canvas.renderAll();`
  - Undo snapshot: `setCanvasJson(JSON.stringify(canvas.toJSON()));`
  - Get dimensions from loaded img: `const w = Math.round(img.width! * (img.scaleX ?? 1)); const h = Math.round(img.height! * (img.scaleY ?? 1));` — but for the toast, show the actual upscaled resolution from the image's natural dimensions: `toast.success(\`Upscaled to ${img.width}x${img.height}\`);` (per D-09)
  - catch: `toast.error(err instanceof Error ? err.message : "Upscale failed")`
  - finally: `setIsProcessing(false)`
- Return `{ handleUpscale }`
