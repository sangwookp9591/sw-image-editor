// Example existing action (removeBackground):
export async function removeBackground(base64Image: string): Promise<{ cdnUrl: string }> {
  await requireAuth();
  const imageBuffer = Buffer.from(base64Image.split(",")[1] || base64Image, "base64");
  const { image } = await generateImage({
    model: fal.image("fal-ai/bria/background/remove"),
    prompt: { text: "", images: [imageBuffer] },
    providerOptions: { fal: { outputFormat: "png", syncMode: true } },
    abortSignal: AbortSignal.timeout(55_000),
  });
  const cdnUrl = await uploadToS3(image.uint8Array, "png", "image/png");
  return { cdnUrl };
}
