1. Create `src/components/theme/theme-provider.tsx` per D-23, D-24:
   ```typescript
   "use client";

   import { ThemeProvider as NextThemesProvider } from "next-themes";

   export function ThemeProvider({ children }: { children: React.ReactNode }) {
     return (
       <NextThemesProvider
         attribute="class"
         defaultTheme="dark"
         enableSystem
         disableTransitionOnChange
       >
         {children}
       </NextThemesProvider>
     );
   }
   ```
   - `attribute="class"` matches Tailwind's `@custom-variant dark (&:is(.dark *))` in globals.css
   - `defaultTheme="dark"` preserves current default experience
   - `enableSystem` respects OS preference
   - `disableTransitionOnChange` prevents flash during toggle

2. Create `src/components/theme/theme-toggle.tsx` per D-25:
   ```typescript
   "use client";

   import { Moon, Sun } from "lucide-react";
   import { useTheme } from "next-themes";
   import { Button } from "@/components/ui/button";

   export function ThemeToggle() {
     const { setTheme, resolvedTheme } = useTheme();
     return (
       <Button
         variant="ghost"
         size="icon-sm"
         onClick={() => setTheme(resolvedTheme === "dark" ? "light" : "dark")}
         aria-label="Toggle theme"
       >
         <Sun className="h-4 w-4 rotate-0 scale-100 transition-all dark:-rotate-90 dark:scale-0" />
         <Moon className="absolute h-4 w-4 rotate-90 scale-0 transition-all dark:rotate-0 dark:scale-100" />
       </Button>
     );
   }
   ```
   - Use `resolvedTheme` (not `theme`) to handle system theme correctly
   - Sun/Moon icons animate based on dark: variant -- Sun visible in light, Moon visible in dark
   - Use `size="icon-sm"` to match other header buttons (same as menu button)
  </action>
